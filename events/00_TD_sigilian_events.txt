## Scandalous Priest

	id = sigilian.1
	title = sigilian.1.t

	}

			loyalty = -5
		}
	}

		name = sigilian.2.b
		add_piety = -0.15
	}
}


## Cultural Conflict with Emperor
country_event = {
	id = sigilian.2
	title = sigilian.2.t
	desc = sigilian.2.desc
	picture = RELIGION_eventPicture
	trigger = {
		NOT = { 
			emperor = {
				primary_culture = ROOT
			}
		}
		NOT = { culture_group = camarian_group }
		NOT = { is_emperor = yes }
		religion = sigilian
	}
	is_triggered_only = yes

	option = {
		name = sigilian.2.a
		add_piety = 0.3
		add_country_modifier = { 
			name = sigilian_2_a_event_modifier
			duration = 5475
		}
	}
	option = {
		name = sigilian.2.b
		add_piety = -0.3
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -20
		}
		add_country_modifier = { 
			name = sigilian_2_b_event_modifier
			duration = 5475
		}
	}
}
