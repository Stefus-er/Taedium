## Cultural Conflict with Emperor
country_event = {
	id = sigilian.2
	title = sigilian.2.t
	desc = sigilian.2.desc
	picture = RELIGION_eventPicture
	trigger = {
		NOT = { 
			emperor = {
				primary_culture = ROOT
			}
		}
		NOT = { culture_group = camarian_group }
	}
	
	mean_time_to_happen = {
		months = 240
		
		modifier = {
			factor = 0.5
			NOT = { emperor = { has_opinion = { who = root value = -10 } } }
		}

	}

	option = {
		name = sigilian.2.a
		add_piety = 0.3
		add_country_modifier = { 
			name = sigilian_2_a_event_modifier
			duration = 5475
		}
	}
	option = {
		name = sigilian.2.b
		add_piety = -0.3
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -20
		}
		add_country_modifier = { 
			name = sigilian_2_b_event_modifier
			duration = 5475
		}
	}
}
