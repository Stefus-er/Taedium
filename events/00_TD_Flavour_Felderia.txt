namespace = felderia


## Formation event
country_event = {
	id = felderia.1
	title = felderia.1.t
	desc = felderia.1.desc

	picture = DIPLOMACY_SIGNING_eventPicture
	
	is_triggered_only = yes

	option = { #Form the country
		name = felderia.1.a
			change_tag = FLD
			on_change_tag_effect = yes
			swap_non_generic_missions = yes
			if = {
				limit = { government_rank = 1 }
				force_set_government_rank = 2
			}
			add_country_modifier = {
				name = "centralization_modifier"
				duration = 7300
			}
			add_prestige = 25
			if = {
				limit = {
					has_custom_ideas = no
				}
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
			set_country_flag = formed_felderia_flag
		add_accepted_culture_or_dip_power = { culture = bogilish dip_reward = yes new_line = yes }
		bogiland_region = {
			NOT = {
				area = nevledi_area
				area = zelnepole_area
				area = ostrowa_area
				area = luchter_area
				area = wedla_area
			}
			add_permanent_claim = ROOT
		}
		modlatz_area = {
			add_permanent_claim = ROOT
		}
		lachsmund_area = {
			add_permanent_claim = ROOT
		}
		altefeld_area = {
			add_permanent_claim = ROOT
		}
		ruhigwald_area = {
			add_permanent_claim = ROOT
		}
		schactheim_area = {
			add_permanent_claim = ROOT
		}
	}
	option = { #Well, if thats what you want..
		name = felderia.1.b
        add_prestige = 30
	}
}
# Felderstadt trade goods event
country_event = {
	id = felderia.2
	title = felderia.2.t
	desc = felderia.1.desc

	picture = DIPLOMACY_SIGNING_eventPicture
	
	is_triggered_only = yes
	trigger = {
		any_owned_province = {
			is_core = ROOT
			has_province_flag = is_felderstadt_flag
		}
	}

	option = { #Glass
		name = felderia.2.a
		random_owned_province = {
			limit = { has_province_flag = is_felderstadt_flag }
			change_trade_goods = glass 
		}
	}
	option = { #Cloth
		name = felderia.2.b
		random_owned_province = {
			limit = { has_province_flag = is_felderstadt_flag }
			change_trade_goods = cloth 
		}
	}
	option = { #Paper
		name = felderia.2.c
		random_owned_province = {
			limit = { has_province_flag = is_felderstadt_flag }
			change_trade_goods = paper 
		}
	}
}
