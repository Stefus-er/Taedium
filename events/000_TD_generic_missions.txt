	}
	option = { #tall
		name = td_missions.2.b
		trigger = {	always = yes }
		set_country_flag = enabled_political_tall
		set_country_flag = enabled_political_missions
		swap_non_generic_missions = yes
	}
	option = { #saelian empire
		name = td_missions.2.c
		trigger = {
			capital_scope = {
				continent = taedium
			}
			religion = sigilian
			government = monarchy

		ai_chance = {

			modifier = {

				NOT = { is_emperor = yes }
			}
		}
	}
