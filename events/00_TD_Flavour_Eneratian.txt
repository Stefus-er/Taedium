		num_of_owned_provinces_with = {
			is_core = ROOT

		}
	}
	
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			random_owned_province = {
                save_event_target_as = TD_Eneratian_Heretic_prov
            }
		}
	}

	option = { #Condemn the Rites
		name = eneratian.3.a 
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 5
		}
		add_fervor = 10

            add_unrest = 15
        }
	}
	option = { #Don't Condemn the Rites
		name = eneratian.3.b
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
		add_fervor = -15
        event_target:TD_Eneratian_Heretic_prov = {
            add_province_modifier = { 
                name = TD_Eneratian_Tolerated_Heresy
                duration = 7300
            }
        }
	}
}

## Bishop opposes heretical Alliance
country_event = {
	id = eneratian.5
	title = eneratian.5.t
	desc = eneratian.5.desc
	picture = RELIGION_eventPicture
	trigger = {
		religion = eneratian
		any_ally = {
			OR = {
				religion = sigilian
				religion = liemarian
			}
		}
	}
	
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			random_ally = {
				limit = {
					OR = {
						religion = sigilian
						religion = liemarian
					}
				}
                save_event_target_as = TD_Eneratian_Heretic_Ally
            }
		}
	}

	option = { #Bribe Him
		name = eneratian.5.a
		add_years_of_income = -1
		random = { 
			chance = 50
			add_country_modifier = { 
				name = TD_Eneratian_Simony_found_out
				duration = 1825
			}
		}
	}
	option = { #Yield in to the demands
		name = eneratian.5.b
		add_fervor = 10
		add_prestige = 5
		event_target:TD_Eneratian_Heretic_Ally = {
			add_opinion = {
				who = ROOT
				modifier = TD_Eneratian_Condemned_heretic_alliance
			}
		}
	}
}

## Church calls for a Holy War!
country_event = {
	id = eneratian.9
	title = eneratian.9.t
	desc = eneratian.9.desc
	picture = RELIGION_eventPicture
	trigger = {
		religion = eneratian
		estate_influence = {
			estate = estate_church
			influence = 80
		}
		any_neighbor_country = { 
			NOT = { religion = eneratian }
		}
	}
	
	is_triggered_only = yes
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					NOT = { religion = eneratian }
				}
                save_event_target_as = TD_Eneratian_holy_war_target
            }
		}
	}

	option = { #DEUS VULT!
		name = eneratian.9.a
		ai_chance = { factor = 1 }
		every_ally = {
			limit = {
				religion = eneratian
			}
			add_casus_belli = {
				target = event_target:TD_Eneratian_holy_war_target
				type = cb_crusade_no_religion_group_restriction
				months = 60
			}
		}
		add_casus_belli = {
			target = event_target:TD_Eneratian_holy_war_target
			type = cb_crusade_no_religion_group_restriction
			months = 60
		}
		event_target:TD_Eneratian_holy_war_target = {
			add_opinion = {
				who = ROOT
				modifier = TD_Eneratian_called_a_holy_war_on_us
			}
		}
	}
	option = { #But he is our ally tho
		name = eneratian.9.b
		ai_chance = { factor = 5 }
		trigger = {
			event_target:TD_Eneratian_holy_war_target = {
				alliance_with = ROOT
			}
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -15
		}
		add_fervor = -50
		every_known_country = {
			limit = {
				religion = eneratian
			}
			add_opinion = {
				who = ROOT
				modifier = TD_Sided_with_heretic_allies
			}
		}
	}
}
