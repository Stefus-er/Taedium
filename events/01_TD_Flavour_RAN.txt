## 5 years of boosts on startup for randwerpen
country_event = {
	id = randwerpen.1
	title = randwerpen.1.t
	desc = randwerpen.1.desc

	picture = SIEGE_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		tag = RAN
	}

	option = {
		name = randwerpen.1.a
		add_country_modifier = { 
			name = randwerpian_deterrence
			duration = 1825
		}
		every_province = {
			limit = {
				OR = {
					owned_by = OOD
					owned_by = LAC
				}
			}
			add_claim = RAN
		}
	}
}

## 5 years of debuffs if country starts conquest outside of intended region.
country_event = {
	id = randwerpen.2
	title = randwerpen.2.t
	desc = randwerpen.2.desc

	picture = LIBERUM_VETO_eventPicture

	trigger = {
		has_country_modifier = randwerpian_deterrence
		num_of_owned_provinces_with = {
			NOT = { region = noordland_region }
			value = 6
		}
	}

	mean_time_to_happen = {
		days = 30
	}
	#is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = randwerpen.2.a
		remove_country_modifier = randwerpian_deterrence
		set_country_flag = RAN_Expanded_outside_Nifthem
		add_country_modifier = { 
			name = randwerpen_misguided_wrath
			duration = 1825
		}
	}
}

country_event = { ## Randwerpen victory
	id = randwerpen.5
	title = randwerpen.5.t
	desc = randwerpen.5.desc
	picture = DIPLOMACY_SIGNING_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	
	option = { ## subjects subjugated
		name = randwerpen.5.a
		every_province = {
			limit = {
				is_nifthem_province = yes
				owned_by = EHR
			}
			cede_province = RAN
		}
		custom_tooltip = NAR_nifthem_new_management_tt
		hidden_effect = {
			add_truce_with = EHR
			EHR = {
				every_subject_country = {
					limit = {
						capital_scope = {
							region = noordland_region
						}
					}
					RAN = {
						create_subject = {
							subject_type = vassal
							who = PREV
						}
					}
					add_truce_with = RAN
					add_truce_with = EHR
				}
			}
		}
	}

	option = { ## subjects freed
		name = randwerpen.5.b
		every_province = {
			limit = {
				is_nifthem_province = yes
				owned_by = EHR
			}
			cede_province = RAN
		}
		custom_tooltip = NAR_nifthem_freedom_tt
		hidden_effect = {
			add_truce_with = EHR
			EHR = {
				every_subject_country = {
					limit = {
						capital_scope = {
							region = noordland_region
						}
					}
					grant_independence = yes
					add_truce_with = RAN
					add_truce_with = EHR
				}
			}
		}
	}

}
