				type = noble_rebels

			}
		}

			limit = {
				is_core = LIH
			}
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		random_owned_province = {
			limit = {
				is_core = LIH
			}
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		add_imperial_influence = 5
	}

		}
	}
}
