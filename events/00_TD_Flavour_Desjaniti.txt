# The Desjaniti events


## The Gsambya Proclamation
country_event = {
	id = desjaniti.1
	title = desjaniti.1.t
	desc = desjaniti.1.desc
	picture = RELIGION_eventPicture
	trigger = {
		tag = BHA
		is_year = 1455
		is_month = 3
		NOT = {
			is_year = 1479
		}
	}
	fire_only_once = yes
	#is_triggered_only = yes

	mean_time_to_happen = {
		days = 7
	}
	
	immediate = {
		set_global_flag = deva_event_happening
	}

	after = {
		clr_global_flag = deva_event_happening
	}
	option = {
		name = desjaniti.1.a
		add_ruler_personality = BHA_living_god
		hidden_effect = {
			every_province = {
				limit = {
					religion = desjaniti
				}
				add_permanent_claim = BHA
			}
		}
		custom_tooltip = BHA_Get_claims_on_desjaniti_provs_tt
		custom_tooltip = BHA_Appoint_next_deva_tt
		ai_chance = {
			factor = 1
		}
	}
}
