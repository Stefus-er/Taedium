					}
				}

			}
		}
		every_subject_country = {
			trigger_switch = {
				on_trigger = government
				theocracy = {
					add_trust = {
						who = ROOT
						value = 20
					}
				}
				monarchy = {
					add_trust = {
						who = ROOT
						value = -20
					}
				}
				republic = {
					add_trust = {
						who = ROOT
						value = 5
					}
				}
			}
		}

			save_global_event_target_as = dead_emperor #Needed for SIS collapse
		}

		SIS = { save_global_event_target_as = emperor } #To make diet compatible
