	conversion_of_convenience = {
		potential = {
			is_lesser_in_union = no
			NOT = { ruler_religion = ROOT }
			OR = {
				has_regency = no
				AND = {
					NOT = { ruler_religion = ROOT }
					has_heir = yes
					NOT = { heir_religion = ROOT }
				}
			}
			OR = {
				ai = no
				has_idea_group = religious_ideas
			}
		}
		allow = {
			stability = 1
			capital_scope = {
				has_owner_religion = yes
			}
		}
		effect = {
			add_stability = -2
			add_legitimacy = 10
			set_ruler_religion = ROOT
			if = {
				limit = {
					has_heir = yes
					NOT = { heir_religion = ROOT }
				}
				set_heir_religion = ROOT
			}
		}
