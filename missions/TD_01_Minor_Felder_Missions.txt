	generic = no
	ai = yes
	has_country_shield = yes

		icon = mission_landsknecht_soldier
		position = 5
		provinces_to_highlight = {  } 

		provinces_to_highlight = {
			OR = {
				AND = {
					culture = felder

						NOT = {
							num_of_owned_provinces_with = {
								is_core = ROOT
								culture = felder
								value = 5
							}
						}
					}
				}
			}
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
			}
		} 
		trigger = { 
			num_of_owned_provinces_with = {
				is_core = ROOT
				culture = felder
				value = 5
			}
		} 

		effect = {
			myslica_area = {
				add_permanent_claim = ROOT
			}
			czernolas_area = {
				add_permanent_claim = ROOT
			}
			schwarzwald_area = {
				add_permanent_claim = ROOT
			}
			neuegrenze_area = {
				add_permanent_claim = ROOT
			}

				add_permanent_claim = ROOT
			}
			schinkhugel_area = {
				add_permanent_claim = ROOT
			}
			add_country_modifier = {
				name = TD_Felder_centralisation_efforts
				duration = 7300
			}
		} 
	}

	TD_Regional_Capital = { #Whole province flag is to avoid duplication of Felderstadt
		icon = mission_early_modern_university
		position = 6
		required_missions = { TD_Felder_Conquest } 
		provinces_to_highlight = {
			if = {
				limit = { NOT = { any_province = { has_province_flag = is_felderstadt_flag } } }
				OR = {
					AND = {
						culture = felder
						root = {
							NOT = {
								any_owned_province = {
									is_core = ROOT
									culture = felder
									development = 16
								}
							}
						}
					}
				}
				OR = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
				}
			}
			else = {
				has_province_flag = is_felderstadt_flag
				OR = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
					NOT = { development = 16 }
				}
			}
		} 

		trigger = {
			if = {
				limit = { NOT = { any_province = { has_province_flag = is_felderstadt_flag } } }
				any_owned_province = {
					is_core = ROOT
					culture = felder
					development = 16
				}
			}
			else = {
				custom_trigger_tooltip = { 
					tooltip = fld_minor_controls_cored_felderstadt_tt
					any_owned_province = {
						is_core = ROOT
						has_province_flag = is_felderstadt_flag
					}
				}
			}
		} 

		effect = {
			custom_tooltip = fld_minor_found_felderstadt_tt
			if = {
				limit = {
					any_owned_province = {
					is_core = ROOT
					culture = felder
					development = 16
					}
				}
				custom_tooltip = BONUS_REWARD_YES
			}
			else = { custom_tooltip = BONUS_REWARD_NO }
				tooltip ={
				change_province_name = Felderstadt
				rename_capital = Felderstadt
				add_center_of_trade_level = 1
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
				}
			custom_tooltip = fld_minor_move_capital_to_felderstadt_tt
			custom_tooltip = fld_minor_felderstadt_choose_trade_good_tt

			custom_tooltip = mission_alternatively_tt

			custom_tooltip = fld_minor_conquered_felderstadt_tt
			if = {
				limit = {
					any_owned_province = {
						is_core = ROOT
						has_province_flag = is_felderstadt_flag
					}
				}
				custom_tooltip = BONUS_REWARD_YES
			}
			else = { custom_tooltip = BONUS_REWARD_NO }
			custom_tooltip = fld_minor_move_capital_to_felderstadt_tt
			tooltip = {
				add_center_of_trade_level = 1
			}
			custom_tooltip = fld_minor_felderstadt_choose_trade_good_tt
			hidden_effect = {
				if = {
					limit = { NOT = { any_province = { has_province_flag = is_felderstadt_flag } } }
					random_owned_province = {
						limit = {
							is_core = ROOT
							culture = felder
							development = 16
						}
						change_province_name = Felderstadt
						rename_capital = Felderstadt
						add_center_of_trade_level = 1
						set_province_flag = is_felderstadt_flag
						add_base_tax = 1
						add_base_production = 1
						add_base_manpower = 1
					}
				}
				else = {
					random_owned_province = {
						limit = { has_province_flag = is_felderstadt_flag }
						move_capital_effect = yes
						add_center_of_trade_level = 1
					}
				}
				country_event = { id = felderia.2 }
			}
		} 
	}
}
