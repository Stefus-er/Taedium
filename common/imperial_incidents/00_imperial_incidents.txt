# default options are 0-based
# In console, can use imperial_incident [key] to immediately begin an incident

incident_nevledian_succession = {
	event = SE_incident.3
	default_option = 0
	can_stop = {
		OR = {
			NOT = {	exists = NEV }
			NEV = {	overlord = { is_part_of_hre = yes }	}
			NEV = { 
				is_part_of_hre = yes
				is_subject = no
			}
		}
	}

	0 = {
		factor = 1
	}
	
	1 = {
		factor = 1
	}
	
	2 = {
		factor = 1
	}
}

incident_regnum_gyrdianicum = {
	event = SE_incident.100
	default_option = 0
	can_stop = {
		always = no
	}
	0 = { #grant the claim
		factor = 1
		modifier = {
			factor = 2
			has_opinion = {
				who = event_target:regnum_gyrdianicum_ET
				value = 50
			}
		}
		modifier = {
			factor = 2
			has_opinion = {
				who = event_target:regnum_gyrdianicum_ET
				value = 100
			}
		}
	}
	1 = {#reject the claim
		factor = 2
		modifier = {
			factor = 5
			is_rival = event_target:regnum_gyrdianicum_ET
		}
		modifier = {
			factor = 2
			NOT = {
				has_opinion = {
					who = event_target:regnum_gyrdianicum_ET
					value = 50
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				has_opinion = {
					who = event_target:regnum_gyrdianicum_ET
					value = 0
				}
			}
		}
		modifier = {
			factor = 2
			is_elector = yes
		}
	}
	2 = {#invite gyrdians instead
		factor = 1
		modifier = {
			factor = 2
			NOT = {
				culture_group = silvestrine_group
			}
		}
	}
}
incident_northern_nobility = {
	event = SE_incident.201
	default_option = 2
	can_stop = {
		emperor = {
			any_owned_province = { culture = middle_magreiner }
			NOT = { primary_culture = middle_magreiner }
		}
	}
	0 = { #decrease northern privileges
		factor = 1
		modifier = {
			factor = 2
			is_emperor = yes
		}
	}
	1 = { #increase southern privileges
		factor = 1
		modifier = {
			factor = 2
			NOT = { primary_culture = middle_magreiner }
			culture_group = silvestrine_group
			is_emperor = no
		}
		modifier = {
			factor = 0.5
			NOT = { government = monarchy } 
		}
	}
	2 = { #create a new kuninsthof
		factor = 1
		modifier = {
			factor = 2
			is_emperor = yes
		}
		modifier = {
			factor = 2
			primary_culture = middle_magreiner
		}
	}
}
