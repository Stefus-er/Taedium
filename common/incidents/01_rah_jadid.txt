incident_ehsani_resurgence = {
	frame = 4
	variable_initial = 7

	potential = {
		has_dlc = "Mandate of Heaven"
		religion = rah_jadid
	}

	trigger = {
		custom_trigger_tooltip = {
			tooltip = rah_jadid_incident_recent_tooltip
			NOT = { is_incident_active = yes }
			OR = {
				NOT = { has_country_flag = ah_jadid_incident_recent }
				had_country_flag = { flag = ah_jadid_incident_recent days = 9125 }
			}
		}
		custom_trigger_tooltip = {
			tooltip = ehsani_happened_tt
			has_global_flag = ehsani_happened_flag
		}
	}

	mean_time_to_happen = {
		months = 30
	}

	immediate_effect = {
		country_event = { id = ehsani_resurgence.2 }
		set_country_flag = active_incident_flag
	}
}
